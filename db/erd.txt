# Relationship Types
#  -    - one to one
#  -<   - one to many
#  >-   - many to one
#  >-<  - many to many
#  -0   - one to zero or one
#  0-   - zero or one to one
#  0-0  - zero or one to zero or one
#  -0<  - one to zero or many
#  >0-  - zero or many to one

request_log
-----------
id integer PK
method_id integer FK >- method.id
url_id integer FK >- url.id
user_agent_id integer NULL FK >- user_agent.id
request_session_id integer NULL FK >0- request_session.id
user_id integer NULL FK >0- user.id
timestamp integer


method
------
id integer PK
method text unique


url
---
id integer PK
url text unique


user_agent
----------
id integer PK
user_agent text unique
count integer
ua_type_id integer NULL FK >0- ua_type.id
ua_bot_id integer NULL FK >0- ua_bot.id


ua_type
-------
id integer PK
name text unique
count integer


ua_bot
------
id integer PK
name text unique
count integer

id integer PK

---------------
id integer PK
language text NULL
timezone text NULL

user
----
id integer PK
username varchar(32) NULL unique
password_hash char(60) NULL
email varchar(320) NULL unique
tel varchar(16) NULL unique
avatar varchar(256) NULL


verification_code
-----------------
id integer PK
passcode char(6)
email varchar(320)
request_time integer
revoke_time integer NULL
match_id integer NULL FK >0- verification_attempt.id
user_id integer NULL FK >0- user.id

id integer PK
passcode char(6)

url text unique

programming_language
--------------------
id integer PK
name text unique


author
------
id integer PK
username text unique

------
id integer PK

repo
----
id integer PK
domain_id integer FK >0- domain.id
author_id integer FK >0- author.id
name text
is_fork boolean NULL
url text unique
desc text NULL

website text NULL
stars integer NULL

repo_keyword

-----------
id integer PK
author_id integer NULL FK >0- author.id
name text unique

desc text NULL
create_time integer NULL
